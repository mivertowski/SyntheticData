\section{Copula Sampling Algorithms}
\label{app:copula_algorithms}

\begin{algorithm}[H]
\caption{Clayton Copula Bivariate Sampling}
\label{alg:clayton}
\SetAlgoLined
\KwIn{Parameter $\theta > 0$, RNG seed}
\KwOut{$(u, v) \in [0,1]^2$ with Clayton dependence}
$u \gets \text{Uniform}(0, 1)$\;
$t \gets \text{Uniform}(0, 1)$\;
$v \gets \left((u^{-\theta} - 1) \cdot t^{-\theta/(\theta+1)} + 1\right)^{-1/\theta}$\;
\Return $(u, v)$\;
\end{algorithm}

\begin{algorithm}[H]
\caption{Gumbel Copula Bivariate Sampling (Marshall-Olkin)}
\label{alg:gumbel}
\SetAlgoLined
\KwIn{Parameter $\theta \geq 1$, RNG seed}
\KwOut{$(u, v) \in [0,1]^2$ with Gumbel dependence}
$S \gets \text{PositiveStable}(1/\theta)$\;
$E_1, E_2 \gets \text{Exponential}(1), \text{Exponential}(1)$\;
$u \gets \exp(-(E_1 / S)^{1/\theta})$\;
$v \gets \exp(-(E_2 / S)^{1/\theta})$\;
\Return $(u, v)$\;
\end{algorithm}

\begin{algorithm}[H]
\caption{Student-$t$ Copula Bivariate Sampling}
\label{alg:student_t}
\SetAlgoLined
\KwIn{Correlation $\rho$, degrees of freedom $\nu$, RNG seed}
\KwOut{$(u, v) \in [0,1]^2$ with Student-$t$ dependence}
$w_1, w_2 \gets \mathcal{N}(0,1), \mathcal{N}(0,1)$\;
$z_1 \gets w_1$\;
$z_2 \gets \rho \cdot w_1 + \sqrt{1 - \rho^2} \cdot w_2$\;
$S \gets \chi^2_\nu / \nu$\;
$t_1 \gets z_1 / \sqrt{S}$\;
$t_2 \gets z_2 / \sqrt{S}$\;
$u \gets F_{t,\nu}(t_1)$\;
$v \gets F_{t,\nu}(t_2)$\;
\Return $(u, v)$\;
\end{algorithm}

\section{COSO Framework Integration}
\label{app:coso}

\system{} implements the COSO 2013 Internal Control---Integrated
Framework with five components and 17 principles:

\begin{center}
\small
\begin{tabular}{clc}
\toprule
\textbf{\#} & \textbf{Component} & \textbf{Principles} \\
\midrule
1 & Control Environment               & 1--5 \\
2 & Risk Assessment                   & 6--9 \\
3 & Control Activities                & 10--12 \\
4 & Information \& Communication      & 13--15 \\
5 & Monitoring Activities             & 16--17 \\
\bottomrule
\end{tabular}
\end{center}

Generated internal controls include 12 transaction-level controls
(C001--C060) and 6 entity-level controls (C070--C081), each mapped to
COSO components, principles, and control scopes.  Maturity levels
(Non-Existent, Ad~Hoc, Repeatable, Defined, Managed, Optimized) are
assigned per control.
